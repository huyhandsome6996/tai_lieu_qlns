2. File _tt: lay_thong_ke

Giống CN02, file thuật toán rất ngắn — chỉ gọi hàm thong_ke() của DSLK (đã học ở bài DSLK).

from loi_thuat_toan.danh_sach_lien_ket import DoublyLinkedList


def lay_thong_ke(danh_sach):
    """
    Trả về dict thống kê toàn bộ danh sách.

    Trả về:
        Dict chứa các key:
        - tong_so           : tổng số mặt hàng
        - dat_nhat          : dict mặt hàng có giá bán cao nhất
        - re_nhat           : dict mặt hàng có giá bán thấp nhất
        - sap_het_hang      : list dict các mặt hàng tồn kho ≤ 5
        - theo_loai         : dict số lượng + tổng tồn theo từng loại
        - tong_gia_tri_kho  : tổng giá trị hàng trong kho (gia_co_ban × ton_kho)
        - tong_ton_kho      : tổng số sản phẩm tồn
    """
    try:
        return danh_sach.thong_ke()
    except Exception as loi:
        print(f'[LOI thong_ke] {loi}')
        return {}

Thuật toán thong_ke() của DSLK

Đây là hàm cốt lõi — nằm trong danh_sach_lien_ket.py. Duyệt qua tất cả Nodes + tích lũy 7 chỉ số:

def thong_ke(self):
    if self.so_luong == 0:
        return {
            'tong_so': 0, 'dat_nhat': None, 're_nhat': None,
            'sap_het_hang': [], 'theo_loai': {},
            'tong_gia_tri_kho': 0, 'tong_ton_kho': 0
        }

    # Khởi tạo biến tích lũy
    dat_nhat = None
    re_nhat = None
    gia_cao_nhat = -1                       # Âm để mọi giá đều lớn hơn
    gia_thap_nhat = float('inf')            # Vô cực để mọi giá đều nhỏ hơn
    sap_het_hang = []
    theo_loai = {}
    tong_gia_tri_kho = 0
    tong_ton_kho = 0

    # Duyệt qua từng Node
    node_hien_tai = self.head
    while node_hien_tai is not None:
        data = node_hien_tai.data
        if hasattr(data, 'tinh_gia_ban'):
            gia_ban = data.tinh_gia_ban()    # ← ĐA HÌNH
            # Cập nhật đắt nhất
            if gia_ban > gia_cao_nhat:
                gia_cao_nhat = gia_ban
                dat_nhat = data
            # Cập nhật rẻ nhất
            if gia_ban < gia_thap_nhat:
                gia_thap_nhat = gia_ban
                re_nhat = data

        if hasattr(data, 'ton_kho'):
            tong_ton_kho += data.ton_kho
            tong_gia_tri_kho += data.gia_co_ban * data.ton_kho
            # Cảnh báo hàng sắp hết khi tồn kho ≤ 5
            if data.ton_kho <= 5:
                sap_het_hang.append(data)

        if hasattr(data, 'loai_hang'):
            loai = data.loai_hang
            if loai not in theo_loai:
                theo_loai[loai] = {'so_luong': 0, 'tong_ton': 0}
            theo_loai[loai]['so_luong'] += 1
            theo_loai[loai]['tong_ton'] += data.ton_kho

        node_hien_tai = node_hien_tai.next

    return {
        'tong_so': self.so_luong,
        'dat_nhat': dat_nhat.chuyen_thanh_dict() if dat_nhat else None,
        're_nhat': re_nhat.chuyen_thanh_dict() if re_nhat else None,
        'sap_het_hang': [item.chuyen_thanh_dict() for item in sap_het_hang],
        'theo_loai': theo_loai,
        'tong_gia_tri_kho': tong_gia_tri_kho,
        'tong_ton_kho': tong_ton_kho
    }

3. File _ui: KhungThongKeUI (LabelFrame)

from tkinter import ttk


class KhungThongKeUI:
    """Khung hiển thị các chỉ số thống kê."""

    def __init__(self, parent):
        # LabelFrame có tiêu đề "Thống kê", padding 10px
        frame = ttk.LabelFrame(parent, text='Thống kê', padding=10)
        frame.pack(fill='x', pady=(0, 5))
        self.frame = frame

        # 5 label hiển thị 5 chỉ số
        self.lbl_tong_so = ttk.Label(frame, text='Tổng số: 0')
        self.lbl_tong_so.pack(anchor='w')

        self.lbl_dat = ttk.Label(frame, text='Đắt nhất: -')
        self.lbl_dat.pack(anchor='w')

        self.lbl_re = ttk.Label(frame, text='Rẻ nhất: -')
        self.lbl_re.pack(anchor='w')

        self.lbl_tri_kho = ttk.Label(frame, text='Tổng giá trị kho: 0đ')
        self.lbl_tri_kho.pack(anchor='w')

        # Foreground='red' để cảnh báo nổi bật
        self.lbl_sap_het = ttk.Label(frame, text='Sắp hết (<5): 0', foreground='red')
        self.lbl_sap_het.pack(anchor='w')

    def cap_nhat_tong_so(self, so):
        self.lbl_tong_so.config(text=f'Tổng số: {so} mặt hàng')

    def cap_nhat_dat_nhat(self, ten, gia):
        self.lbl_dat.config(text=f'Đắt nhất: {ten} ({gia:,.0f}đ)')

    def cap_nhat_re_nhat(self, ten, gia):
        self.lbl_re.config(text=f'Rẻ nhất: {ten} ({gia:,.0f}đ)')

    def cap_nhat_tri_kho(self, tong):
        self.lbl_tri_kho.config(text=f'Tổng giá trị kho: {tong:,.0f}đ')

    def cap_nhat_sap_het(self, so):
        self.lbl_sap_het.config(text=f'Sắp hết (<5): {so}')

    def dat_mac_dinh(self):
        """Reset tất cả label về giá trị mặc định."""
        self.lbl_tong_so.config(text='Tổng số: 0')
        self.lbl_dat.config(text='Đắt nhất: -')
        self.lbl_re.config(text='Rẻ nhất: -')
        self.lbl_tri_kho.config(text='Tổng giá trị kho: 0đ')
        self.lbl_sap_het.config(text='Sắp hết (<5): 0')

Cú pháp mới

3.1. config() để cập nhật widget

Sau khi tạo widget, muốn đổi text/màu/font, dùng config() (hoặc configure() — 2 cái giống nhau):

label = ttk.Label(parent, text='Hello')
# Sau đó muốn đổi text:
label.config(text='World')
# Hoặc đổi foreground:
label.config(foreground='red')
# Hoặc đổi nhiều thứ cùng lúc:
label.config(text='Updated', foreground='blue', font=('Arial', 12, 'bold'))

3.2. foreground + background

  • foreground (viết tắt fg): màu chữ.
  • background (viết tắt bg): màu nền.

Có thể dùng tên màu tiếng Anh ('red', 'blue', 'green') hoặc mã hex ('#FF0000' = đỏ).

3.3. anchor='w' trong pack

Mặc định pack căn giữa. anchor='w' căn lề trái (West). Các giá trị: 'n' (trên), 's' (dưới), 'e' (phải), 'w' (trái), 'center' (giữa).

4. File _dk: DieuKhienThongKe

from .thong_ke_tt import lay_thong_ke


class DieuKhienThongKe:
    def __init__(self, ui_thong_ke, danh_sach):
        self.ui = ui_thong_ke
        self.danh_sach = danh_sach

    def cap_nhat(self):
        """Gọi thuật toán + cập nhật 5 label trên UI."""
        tk = lay_thong_ke(self.danh_sach)
        if not tk:
            self.ui.dat_mac_dinh()
            return

        # Cập nhật từng label
        self.ui.cap_nhat_tong_so(tk.get('tong_so', 0))

        # Mặt hàng đắt nhất
        dat = tk.get('dat_nhat')
        if dat:
            self.ui.cap_nhat_dat_nhat(dat.get('ten_san_pham', '-'),
                                       dat.get('gia_ban', 0))
        else:
            self.ui.lbl_dat.config(text='Đắt nhất: -')

        # Mặt hàng rẻ nhất
        re = tk.get('re_nhat')
        if re:
            self.ui.cap_nhat_re_nhat(re.get('ten_san_pham', '-'),
                                      re.get('gia_ban', 0))
        else:
            self.ui.lbl_re.config(text='Rẻ nhất: -')

        # Tổng giá trị kho
        self.ui.cap_nhat_tri_kho(tk.get('tong_gia_tri_kho', 0))

        # Số mặt hàng sắp hết
        self.ui.cap_nhat_sap_het(len(tk.get('sap_het_hang', [])))

Pattern dict.get(key, default)

Trong code có nhiều dòng tk.get('tong_so', 0) — .get() là method của dict:

# Nếu key có trong dict → trả về giá trị tương ứng
# Nếu key không có → trả về default (tham số 2)
tk = {'tong_so': 10}
tk.get('tong_so', 0)        # 10 (key có)
tk.get('ton_kho', 0)        # 0 (key không có → default)
tk.get('ton_kho')           # None (không truyền default → None)

Khác với tk['ton_kho'] — nếu key không có sẽ raise KeyError. Dùng .get() an toàn hơn.

5. Cú pháp LabelFrame

LabelFrame là Frame có tiêu đề hiển thị trên viền — dùng để nhóm các widget liên quan với nhãn dễ nhìn.

frame = ttk.LabelFrame(parent, text='Thống kê', padding=10)
frame.pack(fill='x', pady=(0, 5))

# Thêm widget con vào LabelFrame
label1 = ttk.Label(frame, text='Item 1')
label1.pack()

label2 = ttk.Label(frame, text='Item 2')
label2.pack()

Khác với Frame thường (không có text), LabelFrame có tiêu đề trên viền — trông gọn gàng hơn khi nhóm chức năng.

💡 Khi nào dùng Frame, LabelFrame, PanedWindow?
  • Frame: nhóm widget khi không cần tiêu đề. Đa dụng nhất.
  • LabelFrame: nhóm widget + cần tiêu đề (như "Thống kê", "Tùy chọn").
  • PanedWindow: chia khung có thể kéo đổi tỷ lệ (như split panel).

6. float('inf') để tìm min

Trong thong_ke() có dòng:

gia_thap_nhat = float('inf')   # Vô cực
# ...
if gia_ban < gia_thap_nhat:
    gia_thap_nhat = gia_ban
    re_nhat = data

Tại sao dùng float('inf')?

Khi tìm giá trị nhỏ nhất, cần khởi tạo biến với giá trị "vô cùng lớn" để mọi giá trị đầu tiên đều nhỏ hơn. Có 2 cách:

# Cách 1: Dùng số rất lớn (khó chọn)
gia_thap_nhat = 9999999999   # Có thể không đủ lớn!

# Cách 2: Dùng float('inf') — chuẩn
gia_thap_nhat = float('inf')   # Vô cực — mọi số đều nhỏ hơn

Tương tự khi tìm max:

gia_cao_nhat = float('-inf')   # Âm vô cực — mọi số đều lớn hơn
# Hoặc:
gia_cao_nhat = -1   # Nếu biết chắc giá ≥ 0
📖 Đa hình trong thong_ke

Trong hàm thong_ke() có dòng:

gia_ban = data.tinh_gia_ban()

Cùng 1 lời gọi, nhưng kết quả khác nhau tùy loại đối tượng:

  • Nếu data là Sach → gia × 1.1
  • Nếu data là TapChi → gia × 1.05
  • v.v.

Đây chính là đa hình OOP — code ngắn gọn mà xử lý được 5 loại.
