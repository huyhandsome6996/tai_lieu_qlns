➕ CN03: Thêm sản phẩm

Đây là chức năng đa hình OOP rõ ràng nhất. Khi user chọn loại hàng (Sách / Tạp chí / Báo giấy / Luận văn / Bản thảo), form tự động hiển thị các trường riêng tương ứng. Lúc lưu, hệ thống tạo đối tượng lớp con phù hợp, gọi hàm tinh_gia_ban() để tính giá bán theo công thức riêng của loại đó.

📖 Mô tả yêu cầu barem

Nhóm 3 — Thêm sản phẩm: User mở form → nhập loại, mã, tên, giá, tồn kho, các trường riêng theo loại → hệ thống validate → tạo đối tượng OOP → lưu vào DSLK + SQLite → ghi lịch sử Undo.

2. Form Thêm dùng chung (FormThemSuaUI)

import tkinter as tk
from tkinter import ttk, messagebox

# Danh sách 5 loại hàng hợp lệ
CAC_LOAI_HANG = ['Sách', 'Tạp chí', 'Báo giấy', 'Luận văn', 'Bản thảo']


class FormThemSuaUI:
    """
    Hộp thoại modal Thêm/Sửa sản phẩm.
    - Khi du_lieu_cu=None: chế độ Thêm mới.
    - Khi du_lieu_cu=dict: chế độ Sửa (đổ dữ liệu cũ vào form, khóa mã + loại).
    """

    def __init__(self, parent, du_lieu_cu=None):
        self.du_lieu_cu = du_lieu_cu
        self.ket_qua = None

        # Tạo cửa sổ Toplevel modal
        self.top = tk.Toplevel(parent)
        self.top.title('Sửa sản phẩm' if du_lieu_cu else 'Thêm sản phẩm')
        self.top.geometry('450x520+450+150')
        self.top.transient(parent)
        self.top.grab_set()

        self._tao_giao_dien()

    def _tao_giao_dien(self):
        """Tạo form với các trường động theo loại hàng."""
        frame = ttk.Frame(self.top, padding=20)
        frame.pack(fill='both', expand=True)

        # === CÁC TRƯỜNG CHUNG ===
        ttk.Label(frame, text='Loại hàng:').grid(row=0, column=0, sticky='e', padx=5, pady=5)
        self.var_loai = tk.StringVar()
        self.combo_loai = ttk.Combobox(frame, textvariable=self.var_loai,
                                        values=CAC_LOAI_HANG, state='readonly', width=22)
        self.combo_loai.grid(row=0, column=1, padx=5, pady=5)
        # Khi user đổi loại → cập nhật lại các trường riêng
        self.combo_loai.bind('<>', lambda e: self._cap_nhat_truong_rieng())

        ttk.Label(frame, text='Mã số:').grid(row=1, column=0, sticky='e', padx=5, pady=5)
        self.entry_ma = ttk.Entry(frame, width=25)
        self.entry_ma.grid(row=1, column=1, padx=5, pady=5)

        ttk.Label(frame, text='Tên sản phẩm:').grid(row=2, column=0, sticky='e', padx=5, pady=5)
        self.entry_ten = ttk.Entry(frame, width=25)
        self.entry_ten.grid(row=2, column=1, padx=5, pady=5)

        ttk.Label(frame, text='Giá cơ bản (đ):').grid(row=3, column=0, sticky='e', padx=5, pady=5)
        self.entry_gia = ttk.Entry(frame, width=25)
        self.entry_gia.grid(row=3, column=1, padx=5, pady=5)

        ttk.Label(frame, text='Tồn kho:').grid(row=4, column=0, sticky='e', padx=5, pady=5)
        self.entry_ton = ttk.Entry(frame, width=25)
        self.entry_ton.grid(row=4, column=1, padx=5, pady=5)

        # === KHUNG TRƯỜNG RIÊNG (động theo loại) ===
        ttk.Separator(frame, orient='horizontal').grid(row=5, column=0, columnspan=2,
                                                        sticky='ew', pady=10)
        ttk.Label(frame, text='Thông tin riêng theo loại:',
                  font=('Arial', 10, 'italic')).grid(row=6, column=0, columnspan=2, sticky='w')

        self.frame_rieng = ttk.Frame(frame)
        self.frame_rieng.grid(row=7, column=0, columnspan=2, sticky='ew', pady=5)
        self.entries_rieng = {}   # Dict lưu các entry trường riêng

        # === NÚT BẤM ===
        frame_nut = ttk.Frame(frame)
        frame_nut.grid(row=8, column=0, columnspan=2, pady=15)
        self.btn_luu = ttk.Button(frame_nut, text='Lưu')
        self.btn_luu.pack(side='left', padx=10)
        ttk.Button(frame_nut, text='Hủy', command=self.top.destroy).pack(side='left', padx=10)

        # === NẾU LÀ SỬA → ĐỔ DỮ LIỆU CŨ VÀO FORM ===
        if self.du_lieu_cu:
            self.var_loai.set(self.du_lieu_cu.get('loai_hang', ''))
            self.combo_loai.config(state='disabled')
            self.entry_ma.insert(0, self.du_lieu_cu.get('ma_so', ''))
            self.entry_ma.config(state='disabled')
            self.entry_ten.insert(0, self.du_lieu_cu.get('ten_san_pham', ''))
            self.entry_gia.insert(0, str(self.du_lieu_cu.get('gia_co_ban', '')))
            self.entry_ton.insert(0, str(self.du_lieu_cu.get('ton_kho', '')))
            self._cap_nhat_truong_rieng()
        else:
            # Chế độ Thêm: mặc định chọn loại đầu tiên
            self.var_loai.set(CAC_LOAI_HANG[0])
            self._cap_nhat_truong_rieng()

    def _cap_nhat_truong_rieng(self):
        """Xóa các entry cũ + tạo entry mới theo loại đang chọn."""
        # Xóa widget cũ trong frame_rieng
        for w in self.frame_rieng.winfo_children():
            w.destroy()
        self.entries_rieng.clear()

        loai = self.var_loai.get()
        # Định nghĩa trường riêng cho từng loại
        truong_rieng = {
            'Sách':      [('tac_gia', 'Tác giả:'), ('nha_xuat_ban', 'Nhà xuất bản:')],
            'Tạp chí':   [('so_phat_hanh', 'Số phát hành:')],
            'Báo giấy':  [('ngay_xuat_ban', 'Ngày xuất bản:')],
            'Luận văn':  [('tac_gia', 'Tác giả:'), ('truong_dai_hoc', 'Trường ĐH:')],
            'Bản thảo':  [('tac_gia', 'Tác giả:'), ('trang_thai', 'Tình trạng:')],
        }
        ds_truong = truong_rieng.get(loai, [])

        # Tạo từng cặp label + entry
        for i, (key, label) in enumerate(ds_truong):
            ttk.Label(self.frame_rieng, text=label).grid(row=i, column=0, sticky='e', padx=5, pady=3)
            e = ttk.Entry(self.frame_rieng, width=25)
            e.grid(row=i, column=1, padx=5, pady=3)
            self.entries_rieng[key] = e
            # Nếu là sửa → đổ giá trị cũ vào
            if self.du_lieu_cu and key in self.du_lieu_cu:
                e.insert(0, str(self.du_lieu_cu[key]))
            elif key == 'trang_thai' and not self.du_lieu_cu:
                # Mặc định "Chưa duyệt" cho Bản thảo mới
                e.insert(0, 'Chưa duyệt')

    def lay_du_lieu(self):
        """Đọc toàn bộ dữ liệu form → trả về dict."""
        du_lieu = {
            'loai_hang':    self.var_loai.get().strip(),
            'ma_so':        self.entry_ma.get().strip(),
            'ten_san_pham': self.entry_ten.get().strip(),
            'gia_co_ban':   self.entry_gia.get().strip(),
            'ton_kho':      self.entry_ton.get().strip(),
        }
        # Thêm trường riêng
        for key, entry in self.entries_rieng.items():
            du_lieu[key] = entry.get().strip()
        return du_lieu

    def bao_loi(self, thong_bao):
        messagebox.showerror('Lỗi', thong_bao, parent=self.top)

    def bao_thanh_cong(self, thong_bao):
        messagebox.showinfo('Thành công', thong_bao, parent=self.top)

    def dong(self):
        self.top.destroy()

3. Cơ chế trường riêng động theo loại

Đây là phần "ma thuật" của form Thêm/Sửa — khi user đổi loại hàng trong combobox, các entry trường riêng tự động thay đổi. Ví dụ:

  • Chọn Sách → hiện 2 entry: Tác giả + Nhà xuất bản
  • Chọn Tạp chí → hiện 1 entry: Số phát hành
  • Chọn Báo giấy → hiện 1 entry: Ngày xuất bản
  • Chọn Luận văn → hiện 2 entry: Tác giả + Trường ĐH
  • Chọn Bản thảo → hiện 2 entry: Tác giả + Tình trạng

3.1. Cấu trúc dữ liệu định nghĩa trường

truong_rieng = {
    'Sách':      [('tac_gia', 'Tác giả:'), ('nha_xuat_ban', 'Nhà xuất bản:')],
    'Tạp chí':   [('so_phat_hanh', 'Số phát hành:')],
    'Báo giấy':  [('ngay_xuat_ban', 'Ngày xuất bản:')],
    'Luận văn':  [('tac_gia', 'Tác giả:'), ('truong_dai_hoc', 'Trường ĐH:')],
    'Bản thảo':  [('tac_gia', 'Tác giả:'), ('trang_thai', 'Tình trạng:')],
}

Đây là dict với key = tên loại, value = list tuple (key_truong, label). Khi user chọn loại, ta tra dict để lấy danh sách trường tương ứng.

3.2. Bind sự kiện combobox

self.combo_loai.bind('<>', lambda e: self._cap_nhat_truong_rieng())

Sự kiện <<ComboboxSelected>> (có 2 dấu < và > ở mỗi bên) là event ảo của Combobox — được kích hoạt khi user chọn 1 item trong dropdown.

3.3. Xóa widget cũ + tạo widget mới

# Xóa widget cũ trong frame_rieng
for w in self.frame_rieng.winfo_children():
    w.destroy()
self.entries_rieng.clear()

# Tạo widget mới theo loại
for i, (key, label) in enumerate(ds_truong):
    ttk.Label(self.frame_rieng, text=label).grid(row=i, column=0, ...)
    e = ttk.Entry(self.frame_rieng, width=25)
    e.grid(row=i, column=1, ...)
    self.entries_rieng[key] = e   # Lưu reference để sau đọc giá trị

winfo_children() trả về list tất cả widget con của frame_rieng. Gọi destroy() cho từng cái để xóa sạch. Sau đó tạo lại widget mới theo loại được chọn.

4. File _tt: them_san_pham (đa hình)

Đây là phần đa hình OOP rõ ràng nhất — tùy loại hàng mà tạo đối tượng lớp con tương ứng:

from loi_thuat_toan.lop_doi_tuong import Sach, TapChi, BaoGiay, LuanVan, BanThao
from loi_thuat_toan.ket_noi_sqlite import luu_mot_san_pham
from loi_thuat_toan.danh_sach_lien_ket import DoublyLinkedList
from loi_thuat_toan.cau_truc_du_lieu import NganXep


def them_san_pham(danh_sach, bo_undoredo, du_lieu):
    """Thêm 1 sản phẩm mới. Trả về (True, msg) hoặc (False, msg)."""
    try:
        # Lấy + làm sạch các trường chung
        loai_hang = du_lieu.get('loai_hang', '').strip()
        ma_so     = du_lieu.get('ma_so', '').strip()
        ten       = du_lieu.get('ten_san_pham', '').strip()
        gia       = float(du_lieu.get('gia_co_ban', 0))   # Ép kiểu float
        ton       = int(du_lieu.get('ton_kho', 0))        # Ép kiểu int

        # Validate cơ bản
        if not ma_so or not ten or not loai_hang:
            return False, 'Vui lòng điền đầy đủ mã, tên, loại.'
        if gia < 0 or ton < 0:
            return False, 'Giá / tồn kho không được âm.'
        # Validate mã không được trùng
        if danh_sach.tim_theo_ma_so(ma_so) is not None:
            return False, f'Mã "{ma_so}" đã tồn tại.'

        # Tạo đối tượng theo loại — OOP ĐA HÌNH
        if loai_hang == 'Sách':
            sp = Sach(ma_so, ten, gia, ton,
                      du_lieu.get('tac_gia', '').strip(),
                      du_lieu.get('nha_xuat_ban', '').strip())
        elif loai_hang == 'Tạp chí':
            sp = TapChi(ma_so, ten, gia, ton,
                        du_lieu.get('so_phat_hanh', '').strip())
        elif loai_hang == 'Báo giấy':
            sp = BaoGiay(ma_so, ten, gia, ton,
                         du_lieu.get('ngay_xuat_ban', '').strip())
        elif loai_hang == 'Luận văn':
            sp = LuanVan(ma_so, ten, gia, ton,
                         du_lieu.get('tac_gia', '').strip(),
                         du_lieu.get('truong_dai_hoc', '').strip(), '')
        elif loai_hang == 'Bản thảo':
            sp = BanThao(ma_so, ten, gia, ton,
                         du_lieu.get('tac_gia', '').strip(),
                         du_lieu.get('trang_thai', 'Chưa duyệt').strip())
        else:
            return False, 'Loại hàng không hợp lệ.'

        # Lưu vào DSLK (RAM) + SQLite (ổ cứng)
        danh_sach.them_vao_cuoi(sp)
        luu_mot_san_pham(sp)

        # Ghi vào Stack Undo: thao tác 'them'
        bo_undoredo['undo'].day_vao({
            'loai': 'them',
            'du_lieu_cu': None,
            'du_lieu_moi': sp.chuyen_thanh_dict()
        })
        # Xóa Redo vì có thao tác mới
        bo_undoredo['redo'].xoa_tat_ca()

        return True, f'Đã thêm "{ten}" vào cửa hàng.'
    except ValueError:
        return False, 'Giá / tồn kho phải là số.'
    except Exception as loi:
        return False, f'Lỗi: {loi}'

Giải thích cú pháp

4.1. Ép kiểu float / int

gia = float(du_lieu.get('gia_co_ban', 0))   # '50000' → 50000.0
ton = int(du_lieu.get('ton_kho', 0))        # '20' → 20

Vì dữ liệu từ Entry luôn là chuỗi (str), cần ép kiểu để lưu vào SQLite đúng kiểu (REAL cho giá, INTEGER cho tồn kho).

Nếu user nhập chữ (VD: 'abc') vào ô giá, float('abc') sẽ raise ValueError → bắt bằng except ValueError → trả về thông báo lỗi thân thiện.

4.2. Đa hình ở đâu?

Khi gọi sp.chuyen_thanh_dict() để ghi vào Undo Stack, hàm này gọi tiếp self.tinh_gia_ban() — và tùy loại đối tượng mà ra công thức khác nhau:

  • Sach: gia_co_ban × 1.1
  • TapChi: gia_co_ban × 1.05
  • BaoGiay: gia_co_ban × 1.0
  • LuanVan: gia_co_ban × 1.2
  • BanThao: gia_co_ban × 1.15

Đó chính là đa hình — cùng một lời gọi hàm (tinh_gia_ban()), nhưng kết quả khác nhau tùy loại đối tượng.

5. File _dk: DieuKhienThemSanPham

from .them_san_pham_tt import them_san_pham
from .them_san_pham_ui import FormThemSuaUI


class DieuKhienThemSanPham:
    def __init__(self, danh_sach, bo_undoredo):
        self.danh_sach = danh_sach
        self.bo_undoredo = bo_undoredo

    def mo_form_them(self, parent):
        """Mở form Thêm modal → chờ user nhập → lưu → đóng form."""
        # Tạo form Thêm (du_lieu_cu=None → chế độ Thêm)
        form = FormThemSuaUI(parent, du_lieu_cu=None)
        # Gắn sự kiện nút Lưu
        form.btn_luu.config(command=lambda: self._xu_ly_luu(form))

        # Chờ đến khi form đóng
        parent.wait_window(form.top)

        # Trả về True nếu có kết quả (đã thêm)
        return form.ket_qua is not None

    def _xu_ly_luu(self, form):
        """Đọc dữ liệu từ form → gọi thuật toán → thông báo kết quả."""
        du_lieu = form.lay_du_lieu()
        ok, msg = them_san_pham(self.danh_sach, self.bo_undoredo, du_lieu)
        if ok:
            form.ket_qua = du_lieu    # Lưu kết quả để caller biết là đã thêm
            form.bao_thanh_cong(msg)
            form.dong()
        else:
            form.bao_loi(msg)         # Giữ form để user sửa

6. Ghi lịch sử Undo khi Thêm

Khi thêm thành công, hệ thống đẩy 1 dict lên Stack Undo:

{
    'loai': 'them',                         # Loại thao tác
    'du_lieu_cu': None,                     # Thêm → không có dữ liệu cũ
    'du_lieu_moi': sp.chuyen_thanh_dict()   # Dữ liệu sản phẩm mới thêm
}

Khi user nhấn Undo (CN08), hệ thống sẽ:

  1. Lấy thao tác này ra khỏi Stack Undo.
  2. Vì loai == 'them' → đảo ngược = XÓA sản phẩm vừa thêm.
  3. Lấy du_lieu_moi['ma_so'] → gọi danh_sach.xoa_node(ma) + xoa_mot_san_pham(ma).
  4. Đẩy thao tác sang Stack Redo (để có thể redo).
⚠️ Xóa Redo sau thao tác mới

Khi có thao tác mới (them/sua/xoa), toàn bộ Redo Stack bị xóa. Vì:

Sau khi Undo A → có thể Redo A. Nhưng nếu thay vì Redo, user thực hiện thao tác mới B, thì lịch sử đã "rẽ nhánh" — A không còn hợp lệ trong nhánh mới.

Đây là quy tắc chuẩn của mọi hệ thống Undo/Redo (Word, Excel, VS Code, v.v.).
