🔍 CN06: Tìm kiếm + Sắp xếp

Chức năng này gộp 2 thao tác: Tìm kiếm (theo tên/mã/loại) và Sắp xếp (theo giá/tên/tồn kho bằng Merge Sort). UI là 1 thanh công cụ ngang gồm ô nhập từ khóa + combobox tiêu chí + nút bấm.

📖 Mô tả yêu cầu barem

Nhóm 6 — Tìm kiếm + Sắp xếp: Khi user nhập từ khóa và chọn tiêu chí (tên/mã/loại), hệ thống duyệt DSLK và trả về các sản phẩm khớp. Khi user nhấn Sắp xếp, hệ thống dùng Merge Sort O(n log n) để sắp xếp DSLK theo tiêu chí (giá bán/tên/tồn kho).

2. File _tt: tim_kiem + sap_xep

from loi_thuat_toan.danh_sach_lien_ket import DoublyLinkedList


def tim_kiem(danh_sach, tu_khoa, tieu_chi='ten'):
    """
    Tìm kiếm sản phẩm theo từ khóa + tiêu chí.

    Tham số:
        danh_sach : DSLK đôi toàn cục.
        tu_khoa   : chuỗi user nhập (đã strip).
        tieu_chi  : 'ten' | 'ma' | 'loại'.

    Trả về:
        List[dict] — danh sách các sản phẩm khớp.
    """
    try:
        tu_khoa = tu_khoa.strip()
        # Nếu từ khóa rỗng → trả về toàn bộ danh sách
        if not tu_khoa:
            return danh_sach.chuyen_thanh_danh_sach()

        if tieu_chi == 'ma':
            # Tìm chính xác theo mã → trả về list có 0 hoặc 1 phần tử
            sp = danh_sach.tim_theo_ma_so(tu_khoa)
            return [sp.chuyen_thanh_dict()] if sp else []

        elif tieu_chi == 'loai':
            # Tìm theo loại hàng (so khớp chứa)
            kq = danh_sach.tim_theo_the_loai(tu_khoa)
        else:
            # Mặc định: tìm theo tên
            kq = danh_sach.tim_kiem(tu_khoa)

        # Chuyển list đối tượng → list dict
        return [sp.chuyen_thanh_dict() for sp in kq if hasattr(sp, 'chuyen_thanh_dict')]
    except Exception as loi:
        print(f'[LOI tim_kiem] {loi}')
        return []


def sap_xep(danh_sach, tieu_chi='gia_ban'):
    """
    Sắp xếp DSLK bằng Merge Sort (O(n log n)) + trả về list dict.

    Tham số:
        danh_sach : DSLK đôi toàn cục.
        tieu_chi  : 'gia_ban' | 'ten_san_pham' | 'ton_kho'.
    """
    try:
        # Gọi hàm sap_xep_tron của DSLK (đã tự cài đặt Merge Sort)
        danh_sach.sap_xep_tron(tieu_chi)
        return danh_sach.chuyen_thanh_danh_sach()
    except Exception as loi:
        print(f'[LOI sap_xep] {loi}')
        return []

3 tiêu chí tìm kiếm

tieu_chi Hàm DSLK gọi So khớp
'ten' (mặc định) tim_kiem(tu_khoa) Tên sản phẩm chứa từ khóa (không phân biệt HOA/thường)
'ma' tim_theo_ma_so(tu_khoa) Khớp chính xác mã số
'loai' tim_theo_the_loai(tu_khoa) Loại hàng chứa từ khóa

3 tiêu chí sắp xếp

tieu_chi Sắp xếp theo Kiểu dữ liệu
'gia_ban' (mặc định) Giá bán (tăng dần) float
'ten_san_pham' Tên sản phẩm (A→Z) str
'ton_kho' Tồn kho (tăng dần) int

3. File _ui: ThanhTimKiemSapXepUI

import tkinter as tk
from tkinter import ttk


class ThanhTimKiemSapXepUI:
    """Thanh công cụ Tìm kiếm + Sắp xếp."""

    def __init__(self, parent):
        # === NHÓM TÌM KIẾM ===
        ttk.Label(parent, text='Tìm:').pack(side='left', padx=2)
        self.entry_tu_khoa = ttk.Entry(parent, width=15)
        self.entry_tu_khoa.pack(side='left', padx=2)

        # Combobox tiêu chí tìm
        self.var_tk_tieu_chi = tk.StringVar(value='ten')
        ttk.Combobox(parent, textvariable=self.var_tk_tieu_chi,
                     values=['ten', 'ma', 'loai'], state='readonly', width=6).pack(side='left', padx=2)

        self.btn_tim = ttk.Button(parent, text='Tìm', width=6)
        self.btn_tim.pack(side='left', padx=2)
        self.btn_lam_moi = ttk.Button(parent, text='↻ Làm mới', width=9)
        self.btn_lam_moi.pack(side='left', padx=2)

        # Separator ngăn cách 2 nhóm
        ttk.Separator(parent, orient='vertical').pack(side='left', fill='y', padx=8)

        # === NHÓM SẮP XẾP ===
        ttk.Label(parent, text='Sắp xếp:').pack(side='left', padx=2)
        self.var_sx_tieu_chi = tk.StringVar(value='gia_ban')
        ttk.Combobox(parent, textvariable=self.var_sx_tieu_chi,
                     values=['gia_ban', 'ten_san_pham', 'ton_kho'],
                     state='readonly', width=12).pack(side='left', padx=2)
        self.btn_sap_xep = ttk.Button(parent, text='Sắp xếp', width=9)
        self.btn_sap_xep.pack(side='left', padx=2)

    def lay_tu_khoa(self):
        return self.entry_tu_khoa.get().strip()

    def lay_tieu_chi_tim(self):
        return self.var_tk_tieu_chi.get()

    def lay_tieu_chi_sap_xep(self):
        return self.var_sx_tieu_chi.get()

    def bind_enter_tim(self, ham_xu_ly):
        """Gắn phím Enter trong ô từ khóa → hàm xử lý tìm."""
        self.entry_tu_khoa.bind('', lambda e: ham_xu_ly())

4. File _dk: DieuKhienTimKiemSapXep

from .tim_kiem_sap_xep_tt import tim_kiem, sap_xep
from chuc_nang.cn02_doc_danh_sach.doc_danh_sach_tt import lay_danh_sach


class DieuKhienTimKiemSapXep:
    def __init__(self, ui_thanh, danh_sach, ui_bang):
        self.ui = ui_thanh
        self.danh_sach = danh_sach
        self.ui_bang = ui_bang

    def gan_su_kien(self):
        """Gắn command cho các nút trên thanh UI."""
        self.ui.btn_tim.config(command=self.xu_ly_tim)
        self.ui.btn_lam_moi.config(command=self.xu_ly_lam_moi)
        self.ui.btn_sap_xep.config(command=self.xu_ly_sap_xep)
        self.ui.bind_enter_tim(self.xu_ly_tim)

    def xu_ly_tim(self):
        """Đọc từ khóa + tiêu chí → gọi thuật toán → đổ kết quả vào bảng."""
        tu_khoa = self.ui.lay_tu_khoa()
        tieu_chi = self.ui.lay_tieu_chi_tim()
        kq = tim_kiem(self.danh_sach, tu_khoa, tieu_chi)
        # Đổ vào bảng
        self.ui_bang.xoa_tat_ca_dong()
        for sp in kq:
            self.ui_bang.them_dong(sp)
        return len(kq)

    def xu_ly_lam_moi(self):
        """Tải lại toàn bộ danh sách từ DSLK."""
        ds = lay_danh_sach(self.danh_sach)
        self.ui_bang.xoa_tat_ca_dong()
        for sp in ds:
            self.ui_bang.them_dong(sp)
        return len(ds)

    def xu_ly_sap_xep(self):
        """Sắp xếp DSLK theo tiêu chí → đổ kết quả vào bảng."""
        tieu_chi = self.ui.lay_tieu_chi_sap_xep()
        kq = sap_xep(self.danh_sach, tieu_chi)
        self.ui_bang.xoa_tat_ca_dong()
        for sp in kq:
            self.ui_bang.them_dong(sp)
        return len(kq)

3 hàm xử lý tương tự nhau

Cả 3 hàm xu_ly_tim, xu_ly_lam_moi, xu_ly_sap_xep đều theo pattern:

  1. Lấy dữ liệu từ UI (từ khóa, tiêu chí).
  2. Gọi thuật toán tương ứng.
  3. Nhận về list dict kết quả.
  4. Xóa toàn bộ dòng trong bảng.
  5. Thêm từng dict vào bảng.

5. Cú pháp Combobox

Combobox là widget dạng dropdown — user click để chọn 1 giá trị từ danh sách có sẵn. Trong đồ án, dùng cho chọn tiêu chí tìm/sắp xếp.

Tạo Combobox cơ bản

# Combobox không có biến liên kết
combo = ttk.Combobox(parent, values=['ten', 'ma', 'loai'], state='readonly', width=6)
combo.pack()

# Lấy giá trị đang chọn
gia_tri = combo.get()

# Set giá trị
combo.set('ma')

Các tham số quan trọng:

  • values: list/tuple các giá trị dropdown.
  • state='readonly': chỉ cho chọn từ list, không cho gõ tự do. Nếu bỏ qua, user có thể gõ chữ vào Combobox.
  • state='disabled': vô hiệu hóa (xám).
  • width=6: độ rộng tính theo ký tự.

Sự kiện Combobox

combo.bind('<>', callback)

<<ComboboxSelected>> là event ảo — kích hoạt khi user chọn 1 item trong dropdown.

6. StringVar trong Combobox

Trong code UI, có dòng:

self.var_tk_tieu_chi = tk.StringVar(value='ten')
ttk.Combobox(parent, textvariable=self.var_tk_tieu_chi,
             values=['ten', 'ma', 'loai'], state='readonly', width=6)

StringVar là gì?

StringVar là biến "đặc biệt" của Tkinter — khi giá trị biến thay đổi, mọi widget liên kết sẽ tự động cập nhật. Đây là cơ chế "data binding" trong Tkinter.

# Tạo StringVar với giá trị mặc định 'ten'
var = tk.StringVar(value='ten')

# Liên kết Combobox với var
combo = ttk.Combobox(parent, textvariable=var, values=['ten', 'ma', 'loai'])

# Khi user chọn 'ma' trong combobox:
# → var tự động thành 'ma' (KHÔNG cần gọi combo.get())

# Đọc giá trị:
print(var.get())   # In ra 'ma'

# Set giá trị:
var.set('loai')
# → combobox tự động hiển thị 'loai'

Tại sao dùng StringVar thay vì combo.get()?

Vì code gọn hơn — không cần lưu reference Combobox, chỉ cần lưu StringVar:

# Cách 1: Không dùng StringVar — phải lưu reference
self.combo = ttk.Combobox(parent, values=['ten', 'ma', 'loai'])
self.combo.pack()
# ...
gia_tri = self.combo.get()    # Phải gọi .get() trên widget

# Cách 2: Dùng StringVar — gọn hơn
self.var = tk.StringVar(value='ten')
ttk.Combobox(parent, textvariable=self.var, values=['ten', 'ma', 'loai']).pack()
# ...
gia_tri = self.var.get()      # Không cần lưu combo reference
💡 Các biến Tkinter tương tự

Ngoài StringVar, còn có:

  • IntVar: biến số nguyên.
  • DoubleVar: biến số thực.
  • BooleanVar: biến boolean.

Dùng cho widget Checkbox, Radiobutton, Scale.

So sánh Tìm vs Làm mới vs Sắp xếp

Nút Hàm gọi Tác dụng Đụng vào DSLK?
Tìm tim_kiem Trả về list dict các sp khớp từ khóa Không — chỉ đọc
Làm mới lay_danh_sach Trả về toàn bộ list dict (như bỏ filter) Không — chỉ đọc
Sắp xếp sap_xep Sắp xếp DSLK (thay đổi thứ tự Nodes) CÓ — sửa thứ tự DSLK
⚠️ Sắp xếp thay đổi DSLK vĩnh viễn

Khi nhấn "Sắp xếp", hàm sap_xep_tron sẽ sắp xếp lại DSLK trong RAM — thay đổi thứ tự các Nodes. Sau đó, mọi thao tác đọc danh sách (CN02, CN06 Tìm) đều thấy thứ tự mới.

Nếu muốn quay lại thứ tự ban đầu, cần thoát app + mở lại (vì SQLite không bị sắp xếp — chỉ DSLK trong RAM bị).
