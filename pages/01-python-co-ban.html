1. Biến và kiểu dữ liệu

Biến giống như cái "hộp" chứa dữ liệu. Bạn đặt tên hộp, rồi bỏ giá trị vào.

# Khai báo biến — KHÔNG cần khai báo kiểu, Python tự đoán
ten = "Nguyễn Văn A"          # str  (chuỗi)
tuoi = 20                     # int  (số nguyên)
chieu_cao = 1.75              # float (số thực)
la_hoc_sinh = True            # bool  (Đúng/Sai)
diem = [8.5, 9.0, 7.5]        # list  (danh sách)

# In kiểu của biến
print(type(ten))              # <class 'str'>
print(type(tuoi))             # <class 'int'>

# Gán lại giá trị (có thể đổi kiểu)
tuoi = "hai mươi"             # giờ thành str
print(type(tuoi))             # <class 'str'>

Trong đồ án, bạn sẽ thấy biến như:

ma_so = "MH001"               # mã mặt hàng (str)
gia_co_ban = 50000            # giá nhập (int)
ton_kho = 20                  # số lượng tồn (int)
loai_hang = "Sách"            # loại hàng (str)

2. Xuất / Nhập dữ liệu

print() để in ra màn hình, input() để đọc từ bàn phím.

# In chuỗi đơn giản
print("Xin chào!")

# In nhiều giá trị — cách nhau bởi dấu phẩy
ten = "Huy"
tuoi = 21
print("Tên:", ten, "Tuổi:", tuoi)
# Kết quả: Tên: Huy Tuổi: 21

# In có định dạng bằng f-string (xem mục 8)
print(f"Tên tôi là {ten}, năm nay {tuoi} tuổi")

# input() luôn trả về chuỗi — cần ép kiểu nếu muốn số
tuoi_nhap = input("Nhập tuổi: ")    # kiểu str
tuoi_so = int(tuoi_nhap)            # ép sang int
print("Tuổi của bạn:", tuoi_so)

3. Câu điều kiện if / elif / else

if = "nếu", elif = "nếu không thì nếu", else = "còn lại".

diem = 7.5

if diem >= 9:
    print("Xuất sắc")
elif diem >= 7:
    print("Khá")
elif diem >= 5:
    print("Trung bình")
else:
    print("Yếu")

# Toán tử so sánh:
# == bằng       != khác
# >  lớn hơn    <  nhỏ hơn
# >= lớn hơn hoặc bằng    <= nhỏ hơn hoặc bằng

# Toán tử logic:
# and (và)   or (hoặc)   not (phủ định)
if diem >= 5 and diem < 7:
    print("Trung bình")

Lưu ý cực kỳ quan trọng: Python dùng khoảng trắng (indent) để phân biệt code nằm trong if hay ngoài if. Thường dùng 4 khoảng trắng.

# ĐÚNG — print nằm trong if (thụt 4 khoảng)
if True:
    print("Trong if")
print("Ngoài if")

# SAI — sẽ báo lỗi IndentationError
# if True:
# print("Thiếu indent")

4. Vòng lặp for / while

Vòng lặp for

Dùng để duyệt qua từng phần tử của 1 danh sách, hoặc lặp N lần.

# Lặp qua list
ds_mon_hoc = ["Toán", "Lý", "Hóa"]
for mon in ds_mon_hoc:
    print(mon)
# Kết quả:
# Toán
# Lý
# Hóa

# Lặp N lần bằng range()
for i in range(5):        # i = 0, 1, 2, 3, 4
    print(f"Lần thứ {i}")

# range(2, 8) → 2, 3, 4, 5, 6, 7
for i in range(2, 8):
    print(i)

# break — thoát vòng lặp
for i in range(10):
    if i == 5:
        break             # dừng khi i = 5
    print(i)              # in 0, 1, 2, 3, 4

# continue — bỏ qua lần lặp hiện tại
for i in range(5):
    if i == 2:
        continue          # bỏ qua i = 2
    print(i)              # in 0, 1, 3, 4

Vòng lặp while

Lặp chừng nào điều kiện còn đúng.

dem = 0
while dem < 5:
    print(f"Đếm: {dem}")
    dem += 1              # tăng dem lên 1 (giống dem = dem + 1)
# In: Đếm: 0, 1, 2, 3, 4
⚠️ Cẩn thận vòng lặp vô hạn

Nếu quên tăng biến đếm trong while, vòng lặp chạy mãi mãi:

# SAI — chạy mãi mãi
dem = 0
while dem < 5:
    print(dem)
    # Quên dem += 1 → dem luôn = 0 → điều kiện luôn đúng

5. List / Dict / Tuple / Set

List — danh sách có thứ tự

# Tạo list
ds = [10, 20, 30, 40, 50]

# Truy cập phần tử (index từ 0)
print(ds[0])              # 10 (phần tử đầu)
print(ds[-1])             # 50 (phần tử cuối, dùng index âm)

# Cắt list (slicing)
print(ds[1:3])            # [20, 30]  (từ index 1 đến 2)
print(ds[:3])             # [10, 20, 30]  (từ đầu đến index 2)
print(ds[2:])             # [30, 40, 50]  (từ index 2 đến cuối)

# Thêm phần tử
ds.append(60)             # thêm vào cuối
ds.insert(0, 5)           # chèn vào vị trí 0

# Xóa phần tử
ds.remove(30)             # xóa phần tử có giá trị 30
ds.pop(0)                 # xóa phần tử ở index 0
del ds[1]                 # xóa phần tử ở index 1

# Độ dài list
print(len(ds))            # số phần tử

# Duyệt list
for x in ds:
    print(x)

Dict — từ điển (key → value)

Dict giống sách từ điển: tra theo "từ" (key) để lấy "nghĩa" (value).

# Tạo dict
san_pham = {
    "ma_so": "MH001",
    "ten": "Truyện Kiều",
    "gia": 50000,
    "ton_kho": 20
}

# Truy cập giá trị theo key
print(san_pham["ten"])              # Truyện Kiều
print(san_pham.get("gia"))          # 50000 (an toàn hơn, không lỗi nếu key không có)
print(san_pham.get("tac_gia", "N/A"))  # N/A (nếu key không có → trả "N/A")

# Thêm / sửa
san_pham["tac_gia"] = "Nguyễn Du"   # thêm key mới
san_pham["gia"] = 55000             # sửa giá trị

# Xóa
del san_pham["ton_kho"]             # xóa key "ton_kho"

# Duyệt dict
for key in san_pham:                # duyệt key
    print(key, "→", san_pham[key])

for key, value in san_pham.items(): # duyệt cả key + value
    print(f"{key}: {value}")

# Kiểm tra key có trong dict không
if "ten" in san_pham:
    print("Có key 'ten'")

Tuple — danh sách KHÔNG thay đổi được

# Tuple giống list nhưng KHÔNG thể sửa
toa_do = (10, 20)
print(toa_do[0])          # 10

# toa_do[0] = 15          # LỖI — tuple không thể gán lại

# Dùng tuple khi có nhiều giá trị cố định
def lay_toa_do():
    return (100, 200)     # trả về tuple

x, y = lay_toa_do()       # unpack tuple
print(x, y)               # 100 200

Set — tập hợp (không trùng lặp)

# Set tự bỏ phần tử trùng
so = {1, 2, 3, 2, 1, 4}
print(so)                 # {1, 2, 3, 4}

# Thêm
so.add(5)

# Kiểm tra tồn tại (nhanh hơn list)
if 3 in so:
    print("Có số 3")

6. Hàm (function)

Hàm là đoạn code đặt tên, có thể gọi lại nhiều lần. Dùng từ khóa def.

# Hàm đơn giản — không tham số, không trả về
def chao():
    print("Xin chào!")

chao()                    # gọi hàm → in "Xin chào!"

# Hàm có tham số
def chao_ten(ten):
    print(f"Chào {ten}!")

chao_ten("Huy")           # Chào Huy!
chao_ten("Tú")            # Chào Tú!

# Hàm có giá trị mặc định cho tham số
def chao_loi(ten, loi="Xin chào"):
    print(f"{loi}, {ten}!")

chao_loi("Huy")                    # Xin chào, Huy!
chao_loi("Tú", "Hi")               # Hi, Tú!

# Hàm trả về giá trị (return)
def tinh_tong(a, b):
    return a + b

ket_qua = tinh_tong(3, 5)  # ket_qua = 8

# Hàm trả về nhiều giá trị (thực ra là trả tuple)
def tinh(a, b):
    return a + b, a - b, a * b

tong, hieu, tich = tinh(10, 3)
print(tong, hieu, tich)    # 13 7 30

# Hàm có docstring (tài liệu)
def tinh_binh_phuong(x):
    """
    Tính bình phương của x.
    Tham số: x (int hoặc float)
    Trả về: x * x
    """
    return x * x

# Truy cập docstring
print(tinh_binh_phuong.__doc__)

7. Xử lý ngoại lệ try / except

Khi code có thể gây lỗi (chia cho 0, file không tồn tại...), dùng try/except để bắt lỗi mà không sập chương trình.

# Cơ bản
try:
    so = int(input("Nhập số: "))
    print(10 / so)
except:
    print("Có lỗi xảy ra!")

# Bắt lỗi cụ thể
try:
    so = int("abc")              # lỗi ValueError
    f = open("khong_co.txt")     # lỗi FileNotFoundError
except ValueError:
    print("Không phải số nguyên!")
except FileNotFoundError:
    print("File không tồn tại!")
except Exception as loi:         # bắt mọi lỗi còn lại
    print(f"Lỗi khác: {loi}")

# Có thể có finally — chạy dù có lỗi hay không
try:
    f = open("data.txt")
    data = f.read()
except FileNotFoundError:
    print("File không có")
finally:
    # Dù thành công hay lỗi, đều đóng file
    if 'f' in dir():
        f.close()

# Raise — tự ném lỗi
def kiem_tra_tuoi(tuoi):
    if tuoi < 0:
        raise ValueError("Tuổi không được âm!")
    return tuoi

try:
    kiem_tra_tuoi(-5)
except ValueError as e:
    print(e)                     # Tuổi không được âm!
ℹ️ Trong đồ án

Đồ án dùng try/except để bọc 7 thao tác chính: Thêm, Sửa, Xóa, Undo, Redo, Giỏ hàng, Thanh toán. Nếu lỗi → in ra + không sập app.

File SQLite cũng bọc try/except sqlite3.Error để bắt lỗi DB.

8. f-string (chuỗi định dạng)

f-string là cách dễ nhất để ghép biến vào chuỗi. Đặt chữ f trước chuỗi, rồi dùng {biến} để chèn giá trị.

ten = "Huy"
tuoi = 21
diem = 8.567

# Cơ bản
print(f"Tên: {ten}, Tuổi: {tuoi}")
# Tên: Huy, Tuổi: 21

# Format số thập phân
print(f"Điểm: {diem:.2f}")        # 8.57 (2 chữ số thập phân)
print(f"Điểm: {diem:.0f}")        # 9 (làm tròn thành 9)

# Tính toán trong f-string
print(f"Tổng: {2 + 3}")           # Tổng: 5

# Gọi hàm trong f-string
def lay_ho(ten):
    return ten.split()[0]

print(f"Họ: {lay_ho('Nguyễn Văn A')}")  # Họ: Nguyễn

# Căn lề
print(f"{'Tên':<15} {'Tuổi':>5}")   # Tên             Tuổi
print(f"{ten:<15} {tuoi:>5}")       # Huy                21

9. Import module

Module là file .py chứa code. Khi muốn dùng code từ file khác, ta import nó.

# Cách 1: import cả module
import math
print(math.pi)              # 3.141592653589793
print(math.sqrt(16))        # 4.0

# Cách 2: import 1 hàm/lớp cụ thể (khuyên dùng)
from math import sqrt, pi
print(sqrt(25))             # 5.0
print(pi)                   # 3.141592653589793

# Cách 3: đổi tên khi import (tránh trùng tên)
import math as m
print(m.pi)

from math import sqrt as can_bac_2
print(can_bac_2(9))         # 3.0

# Import từ file khác trong cùng project
# Giả sử có file: lop_doi_tuong.py chứa lớp Sach
from lop_doi_tuong import Sach

sach = Sach("MH001", "Truyện Kiều", 50000, 20, "Nguyễn Du", "NXB Văn học")

# Import từ folder con
# Cấu trúc: loi_thuat_toan/lop_doi_tuong.py
from loi_thuat_toan.lop_doi_tuong import Sach, TapChi, BaoGiay

# Import nhiều thứ cùng lúc
from loi_thuat_toan.lop_doi_tuong import (
    Sach,
    TapChi,
    BaoGiay,
    LuanVan,
    BanThao
)

10. hasattr / getattr / setattr

3 hàm này dùng để kiểm tra / lấy / gán thuộc tính của đối tượng bằng tên (chuỗi).

# Tạo lớp đơn giản
class Nguoi:
    def __init__(self, ten, tuoi):
        self.ten = ten
        self.tuoi = tuoi

nguoi = Nguoi("Huy", 21)

# hasattr(obj, "ten_thuoc_tinh") — kiểm tra có thuộc tính không
print(hasattr(nguoi, "ten"))       # True
print(hasattr(nguoi, "dia_chi"))   # False

# getattr(obj, "ten", gia_tri_mac_dinh)
print(getattr(nguoi, "ten"))                # Huy
print(getattr(nguoi, "dia_chi", "N/A"))     # N/A (không lỗi)

# setattr(obj, "ten", gia_tri) — gán thuộc tính
setattr(nguoi, "dia_chi", "Hà Nội")
print(nguoi.dia_chi)                # Hà Nội

# Tương đương:
# nguoi.dia_chi = "Hà Nội"
ℹ️ Trong đồ án

Đồ án dùng hasattr rất nhiều trong DSLK để kiểm tra đối tượng có thuộc tính cần thiết không trước khi gọi:

# Trong danh_sach_lien_ket.py
if hasattr(node_hien_tai.data, 'ma_so') and node_hien_tai.data.ma_so == ma_so:
    # ...

# Trong ket_noi_sqlite.py — dùng getattr với default None
con_tro.execute('INSERT ...', (
    san_pham.ma_so,
    getattr(san_pham, 'tac_gia', None),       # nếu không có → None
    getattr(san_pham, 'so_phat_hanh', None),  # nếu không có → None
))
