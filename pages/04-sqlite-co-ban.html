💾 Bài 4: SQLite cơ bản

SQLite là cơ sở dữ liệu nhẹ, lưu trong 1 file (.sqlite3). Không cần cài server — Python có sẵn module sqlite3.

Đây là phần Ngọ phụ trách (vì đã học SQLite + Stack + Queue).

📑 Mục lục
  1. Kết nối + tạo bảng
  2. INSERT — thêm dữ liệu
  3. SELECT — đọc dữ liệu
  4. UPDATE — sửa dữ liệu
  5. DELETE — xóa dữ liệu
  6. Parameterized query — chống SQL Injection
  7. try/except/finally cho SQLite
  8. Phân tích file ket_noi_sqlite.py

1. Kết nối + tạo bảng

import sqlite3

# 1. Kết nối tới file DB (nếu file không có → tự tạo mới)
ket_noi = sqlite3.connect("test.db")

# 2. Tạo con trỏ (cursor) để thực thi SQL
con_tro = ket_noi.cursor()

# 3. Tạo bảng
con_tro.execute('''
    CREATE TABLE IF NOT EXISTS hoc_sinh (
        id    INTEGER PRIMARY KEY AUTOINCREMENT,
        ten   TEXT NOT NULL,
        tuoi  INTEGER,
        diem  REAL
    )
''')

# 4. Lưu thay đổi (commit)
ket_noi.commit()

# 5. Đóng kết nối
ket_noi.close()

Giải thích từng dòng

  • sqlite3.connect("test.db") — mở file test.db. Nếu chưa có → tạo file mới. Trả về đối tượng kết nối.
  • .cursor() — tạo "con trỏ" — công cụ để chạy SQL.
  • con_tro.execute("SQL...") — chạy 1 câu SQL.
  • CREATE TABLE IF NOT EXISTS — tạo bảng nếu chưa có (tránh lỗi khi bảng đã tồn tại).
  • INTEGER PRIMARY KEY AUTOINCREMENT — số nguyên, khóa chính, tự tăng.
  • TEXT — chuỗi. INTEGER — số nguyên. REAL — số thực.
  • NOT NULL — bắt buộc nhập, không được để trống.
  • ket_noi.commit() — LƯU thay đổi xuống file. Quên commit = mất dữ liệu khi đóng!
  • ket_noi.close() — đóng kết nối, giải phóng tài nguyên.

2. INSERT — thêm dữ liệu

import sqlite3

ket_noi = sqlite3.connect("test.db")
con_tro = ket_noi.cursor()

# Cách 1: nối chuỗi (NGUY HIỂM — dễ bị SQL Injection)
# KHÔNG nên dùng!
# ten = "Robert'); DROP TABLE hoc_sinh;--"
# con_tro.execute(f"INSERT INTO hoc_sinh (ten, tuoi, diem) VALUES ('{ten}', 20, 8.5)")

# Cách 2: PARAMETERIZED QUERY (an toàn — dùng dấu ?)
con_tro.execute(
    "INSERT INTO hoc_sinh (ten, tuoi, diem) VALUES (?, ?, ?)",
    ("Nguyễn Văn A", 20, 8.5)
)

# Thêm nhiều dòng cùng lúc bằng executemany
ds_hoc_sinh = [
    ("Trần Thị B", 21, 9.0),
    ("Lê Văn C", 19, 7.5),
    ("Phạm Thị D", 22, 8.0),
]
con_tro.executemany(
    "INSERT INTO hoc_sinh (ten, tuoi, diem) VALUES (?, ?, ?)",
    ds_hoc_sinh
)

ket_noi.commit()                  # LƯU thay đổi
ket_noi.close()
🚨 SQL Injection là gì?

Nếu user nhập tên Robert'); DROP TABLE hoc_sinh;-- vào ô nhập, và bạn dùng nối chuỗi để tạo SQL — hacker có thể xóa toàn bộ bảng của bạn!

Giải pháp: luôn dùng dấu ? (parameterized query). SQLite tự thoát ký tự đặc biệt → an toàn 100%.

5. DELETE — xóa dữ liệu

import sqlite3

ket_noi = sqlite3.connect("test.db")
con_tro = ket_noi.cursor()

# Xóa theo điều kiện
con_tro.execute("DELETE FROM hoc_sinh WHERE id = ?", (1,))

# ⚠️ Lưu ý: (1,) có dấu phẩy — vì đây là tuple 1 phần tử
# Nếu viết (1) → Python hiểu là số 1, không phải tuple → LỖI

ket_noi.commit()
print(f"Đã xóa {con_tro.rowcount} dòng.")

# Xóa tất cả (CẨN THẬN!)
# con_tro.execute("DELETE FROM hoc_sinh")

ket_noi.close()
⚠️ Tuple 1 phần tử cần dấu phẩy

Sai: (1) → Python hiểu là số 1 (trong ngoặc đơn toán học).

Đúng: (1,) → tuple chứa 1 phần tử 1.

SQLite cần tuple để thay dấu ?, nên phải có dấu phẩy.

6. Parameterized query — chống SQL Injection

Quy tắc vàng: KHÔNG BAO GIỜ nối chuỗi để tạo SQL. Luôn dùng dấu ?.

# ❌ SAI — nguy cơ SQL Injection
ten = input("Nhập tên: ")
sql = f"SELECT * FROM hoc_sinh WHERE ten = '{ten}'"
con_tro.execute(sql)

# ✅ ĐÚNG — parameterized query
ten = input("Nhập tên: ")
con_tro.execute("SELECT * FROM hoc_sinh WHERE ten = ?", (ten,))

# Nhiều tham số
con_tro.execute(
    "SELECT * FROM hoc_sinh WHERE tuoi >= ? AND diem >= ?",
    (18, 8.0)
)

# Tên bảng/cột không thể dùng ? — phải nối chuỗi AN TOÀN
# (chỉ khi giá trị đến từ code cố định, không từ user)
bang = "hoc_sinh"
con_tro.execute(f"SELECT * FROM {bang}")

7. try/except/finally cho SQLite

Mọi thao tác SQLite đều nên bọc try/except/finally:

import sqlite3

ket_noi = None                              # khai báo trước để finally truy cập được
try:
    ket_noi = sqlite3.connect("test.db")
    con_tro = ket_noi.cursor()

    con_tro.execute("INSERT INTO hoc_sinh (ten, tuoi) VALUES (?, ?)", ("Test", 20))
    ket_noi.commit()

except sqlite3.IntegrityError as loi:
    # Lỗi ràng buộc (vd: trùng khóa chính)
    print(f"Trùng khóa: {loi}")

except sqlite3.Error as loi:
    # Lỗi SQLite khác
    print(f"Lỗi DB: {loi}")
    raise                                   # ném lỗi lên để xử lý tiếp

finally:
    # LUÔN đóng kết nối — dù thành công hay lỗi
    if ket_noi:
        ket_noi.close()

Các loại lỗi SQLite thường gặp

LỗiNguyên nhân
sqlite3.IntegrityErrorTrùng khóa chính / vi phạm ràng buộc NOT NULL / FOREIGN KEY
sqlite3.OperationalErrorBảng không tồn tại, cú pháp SQL sai, DB bị khóa
sqlite3.ProgrammingErrorSố tham số ? không khớp số giá trị truyền vào
sqlite3.ErrorLỗi SQLite tổng quát (cha của tất cả lỗi trên)

8.2. Tạo bảng

def tao_bang_neu_chua_co():
    """Tạo 2 bảng nếu chưa tồn tại."""
    ket_noi = None                                  # khai báo trước cho finally
    try:
        duong_dan = lay_duong_dan_db()
        ket_noi = sqlite3.connect(duong_dan)         # mở kết nối
        con_tro = ket_noi.cursor()

        # Bảng mặt hàng — 12 cột
        # Các trường riêng (tac_gia, so_phat_hanh, ...) có thể NULL
        # vì không phải loại nào cũng có
        con_tro.execute('''
            CREATE TABLE IF NOT EXISTS bang_mat_hang (
                ma_so         TEXT PRIMARY KEY,      -- khóa chính
                loai_hang     TEXT NOT NULL,
                ten_san_pham  TEXT NOT NULL,
                gia_co_ban    REAL NOT NULL,
                ton_kho       INTEGER NOT NULL,
                tac_gia       TEXT,                   -- Sách/Luận văn/Bản thảo
                nha_xuat_ban  TEXT,                   -- Sách
                so_phat_hanh  TEXT,                   -- Tạp chí
                ngay_xuat_ban TEXT,                   -- Báo giấy
                truong_dai_hoc TEXT,                  -- Luận văn
                nam_bao_ve    TEXT,                   -- Luận văn
                tinh_trang    TEXT                    -- Bản thảo
            )
        ''')

        # Bảng người dùng — 4 cột
        con_tro.execute('''
            CREATE TABLE IF NOT EXISTS bang_nguoi_dung (
                ten_dang_nhap  TEXT PRIMARY KEY,
                mat_khau_hash  TEXT NOT NULL,
                ho_ten         TEXT NOT NULL,
                vai_tro        TEXT NOT NULL DEFAULT 'nhan_vien'
            )
        ''')

        ket_noi.commit()                             # LƯU thay đổi
    except sqlite3.Error as loi:
        print(f'[LOI DATABASE] Loi khi tao bang: {str(loi)}')
        raise                                        # ném lỗi lên
    finally:
        # LUÔN đóng kết nối
        if ket_noi:
            ket_noi.close()

8.3. Tải dữ liệu từ SQLite vào DSLK

def tai_du_lieu(danh_sach_lien_ket):
    """Đọc tất cả mặt hàng từ SQLite → tạo đối tượng OOP → thêm vào DSLK."""
    # Lazy import để tránh phụ thuộc vòng
    from loi_thuat_toan.lop_doi_tuong import Sach, TapChi, BaoGiay, LuanVan, BanThao

    tao_bang_neu_chua_co()                           # đảm bảo bảng có
    ket_noi = None
    try:
        ket_noi = sqlite3.connect(lay_duong_dan_db())
        con_tro = ket_noi.cursor()

        # Lấy tất cả hàng
        con_tro.execute('SELECT * FROM bang_mat_hang')
        cac_hang = con_tro.fetchall()                # list các tuple

        # Lấy tên cột
        ten_cot = [mo_ta[0] for mo_ta in con_tro.description]

        # Duyệt từng hàng → tạo đối tượng OOP → thêm vào DSLK
        for hang in cac_hang:
            du_lieu = dict(zip(ten_cot, hang))       # ép tuple → dict

            ma_so = du_lieu['ma_so']
            loai_hang = du_lieu['loai_hang']
            ten_san_pham = du_lieu['ten_san_pham']
            gia_co_ban = du_lieu['gia_co_ban']
            ton_kho = du_lieu['ton_kho']

            # Tùy loại hàng → khởi tạo đối tượng OOP khác nhau (ĐA HÌNH)
            if loai_hang == 'Sách':
                san_pham = Sach(
                    ma_so, ten_san_pham, gia_co_ban, ton_kho,
                    du_lieu.get('tac_gia') or '',
                    du_lieu.get('nha_xuat_ban') or ''
                )
            elif loai_hang == 'Tạp chí':
                san_pham = TapChi(
                    ma_so, ten_san_pham, gia_co_ban, ton_kho,
                    du_lieu.get('so_phat_hanh') or ''
                )
            elif loai_hang == 'Báo giấy':
                san_pham = BaoGiay(
                    ma_so, ten_san_pham, gia_co_ban, ton_kho,
                    du_lieu.get('ngay_xuat_ban') or ''
                )
            elif loai_hang == 'Luận văn':
                san_pham = LuanVan(
                    ma_so, ten_san_pham, gia_co_ban, ton_kho,
                    du_lieu.get('tac_gia') or '',
                    du_lieu.get('truong_dai_hoc') or '',
                    du_lieu.get('nam_bao_ve') or ''
                )
            elif loai_hang == 'Bản thảo':
                san_pham = BanThao(
                    ma_so, ten_san_pham, gia_co_ban, ton_kho,
                    du_lieu.get('tac_gia') or '',
                    du_lieu.get('tinh_trang') or ''
                )
            else:
                continue                              # loại lạ → bỏ qua

            danh_sach_lien_ket.them_vao_cuoi(san_pham)

    except sqlite3.Error as loi:
        print(f'[LOI DATABASE] Loi khi tai du lieu: {str(loi)}')
        raise
    finally:
        if ket_noi:
            ket_noi.close()

8.4. INSERT — luu_mot_san_pham

def luu_mot_san_pham(san_pham):
    """Lưu 1 sản phẩm mới vào SQLite."""
    ket_noi = None
    try:
        ket_noi = sqlite3.connect(lay_duong_dan_db())
        con_tro = ket_noi.cursor()

        # 12 dấu ? tương ứng 12 giá trị
        con_tro.execute('''
            INSERT INTO bang_mat_hang (
                ma_so, loai_hang, ten_san_pham, gia_co_ban, ton_kho,
                tac_gia, nha_xuat_ban, so_phat_hanh, ngay_xuat_ban,
                truong_dai_hoc, nam_bao_ve, tinh_trang
            )
            VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
        ''', (
            san_pham.ma_so,
            san_pham.loai_hang,
            san_pham.ten_san_pham,
            san_pham.gia_co_ban,
            san_pham.ton_kho,
            # getattr(obj, attr, default) — trả default nếu đối tượng không có thuộc tính
            # Sach có tac_gia, nhưng TapChi không có → getattr trả None
            getattr(san_pham, 'tac_gia', None),
            getattr(san_pham, 'nha_xuat_ban', None),
            getattr(san_pham, 'so_phat_hanh', None),
            getattr(san_pham, 'ngay_xuat_ban', None),
            getattr(san_pham, 'truong_dai_hoc', None),
            getattr(san_pham, 'nam_bao_ve', None),
            getattr(san_pham, 'tinh_trang', None)
        ))
        ket_noi.commit()                             # LƯU xuống file
    except sqlite3.IntegrityError as loi:
        # Thường do trùng ma_so (khóa chính)
        print(f'[LOI DATABASE] Ma so da ton tai: {str(loi)}')
        raise
    except sqlite3.Error as loi:
        print(f'[LOI DATABASE] Loi khi luu: {str(loi)}')
        raise
    finally:
        if ket_noi:
            ket_noi.close()

8.5. UPDATE — cap_nhat_san_pham

def cap_nhat_san_pham(san_pham):
    """Cập nhật sản phẩm theo ma_so."""
    ket_noi = None
    try:
        ket_noi = sqlite3.connect(lay_duong_dan_db())
        con_tro = ket_noi.cursor()
        con_tro.execute('''
            UPDATE bang_mat_hang SET
                loai_hang = ?, ten_san_pham = ?, gia_co_ban = ?, ton_kho = ?,
                tac_gia = ?, nha_xuat_ban = ?, so_phat_hanh = ?, ngay_xuat_ban = ?,
                truong_dai_hoc = ?, nam_bao_ve = ?, tinh_trang = ?
            WHERE ma_so = ?
        ''', (
            san_pham.loai_hang, san_pham.ten_san_pham, san_pham.gia_co_ban, san_pham.ton_kho,
            getattr(san_pham, 'tac_gia', None),
            getattr(san_pham, 'nha_xuat_ban', None),
            getattr(san_pham, 'so_phat_hanh', None),
            getattr(san_pham, 'ngay_xuat_ban', None),
            getattr(san_pham, 'truong_dai_hoc', None),
            getattr(san_pham, 'nam_bao_ve', None),
            getattr(san_pham, 'tinh_trang', None),
            san_pham.ma_so                          # điều kiện WHERE
        ))
        ket_noi.commit()
        # rowcount = số hàng bị ảnh hưởng (1 = OK, 0 = không tìm thấy)
        return con_tro.rowcount > 0
    except sqlite3.Error as loi:
        print(f'[LOI DATABASE] Loi khi cap nhat: {str(loi)}')
        raise
    finally:
        if ket_noi:
            ket_noi.close()

                INSERT INTO bang_mat_hang VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
            ''', sp)
        ket_noi.commit()
        print('[KHOI TAO] Da them 5 san pham mau.')
    except sqlite3.Error as loi:
        print(f'[LOI DATABASE] Loi khi tao du lieu mau: {str(loi)}')
    finally:
        if ket_noi:
            ket_noi.close()
