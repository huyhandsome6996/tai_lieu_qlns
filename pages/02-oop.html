1. Lớp (class) và Đối tượng (object) là gì?

Lớp (class) giống như bản vẽ thiết kế một loại đồ vật. Đối tượng (object) là sản phẩm thực tế được tạo ra từ bản vẽ đó.

Ví dụ: Lớp "Chiếc xe" là bản vẽ. Chiếc xe của bạn, chiếc xe của tôi là các đối tượng được tạo ra từ lớp đó.

# Định nghĩa lớp
class Cho:
    # Lớp này có 2 thuộc tính: ten, tuoi
    def __init__(self, ten, tuoi):
        self.ten = ten
        self.tuoi = tuoi

    # Lớp này có 1 phương thức: sua
    def sua(self):
        print(f"{self.ten} sắp: Gâu gâu!")

# Tạo 2 đối tượng từ lớp Cho
cho_1 = Cho("Milu", 3)      # tạo chó tên Milu, 3 tuổi
cho_2 = Cho("Bull", 5)      # tạo chó tên Bull, 5 tuổi

# Gọi phương thức
cho_1.sua()                  # Milu sắp: Gâu gâu!
cho_2.sua()                  # Bull sắp: Gâu gâu!

# Truy cập thuộc tính
print(cho_1.ten)             # Milu
print(cho_2.tuoi)            # 5
ℹ️ Giải thích self

self = "chính đối tượng này". Khi gọi cho_1.sua(), Python tự truyền cho_1 vào self. Nên trong hàm sua, self.ten chính là cho_1.ten = "Milu".

Đừng quên self ở tham số đầu tiên của mọi hàm trong lớp!

2. Hàm __init__ — Khởi tạo đối tượng

__init__ (2 gạch dưới + init + 2 gạch dưới) là hàm tạo. Nó tự động chạy khi bạn tạo đối tượng mới.

class HocSinh:
    def __init__(self, ten, tuoi, lop):
        # self.xxx = yyy → gán giá trị yyy vào thuộc tính xxx
        self.ten = ten            # tạo thuộc tính ten
        self.tuoi = tuoi          # tạo thuộc tính tuoi
        self.lop = lop            # tạo thuộc tính lop

    def gioi_thieu(self):
        print(f"Tôi là {self.ten}, {self.tuoi} tuổi, học lớp {self.lop}")

# Tạo đối tượng — tự động chạy __init__
hs = HocSinh("Huy", 21, "12A1")
hs.gioi_thieu()              # Tôi là Huy, 21 tuổi, học lớp 12A1

3. Phương thức (method) — hàm trong lớp

Phương thức = hàm đặt trong lớp. Luôn có self ở tham số đầu.

class MayTinh:
    def __init__(self, hang, ram):
        self.hang = hang
        self.ram = ram

    # Phương thức không tham số (ngoài self)
    def thong_tin(self):
        print(f"{self.hang} - RAM {self.ram}GB")

    # Phương thức có tham số
    def nang_cap_ram(self, them_gb):
        self.ram += them_gb
        print(f"Đã thêm {them_gb}GB. RAM mới: {self.ram}GB")

    # Phương thức trả về giá trị
    def lay_ram(self):
        return self.ram

mt = MayTinh("Dell", 8)
mt.thong_tin()                       # Dell - RAM 8GB
mt.nang_cap_ram(8)                   # Đã thêm 8GB. RAM mới: 16GB
print("RAM hiện tại:", mt.lay_ram()) # RAM hiện tại: 16

4. Kế thừa (inheritance) — lớp con kế thừa lớp cha

Kế thừa = lớp con nhận lại tất cả thuộc tính + phương thức của lớp cha, rồi có thể thêm cái của riêng mình.

Ví dụ: DongVat là lớp cha, Cho và Meo là 2 lớp con. Cả 2 đều có thuộc tính "ten" (kế thừa từ DongVat), nhưng tiếng kêu khác nhau.

# Lớp cha
class DongVat:
    def __init__(self, ten):
        self.ten = ten

    def an(self):
        print(f"{self.ten} đang ăn...")

# Lớp con — kế thừa từ DongVat
class Cho(DongVat):                  # ← (DongVat) = kế thừa
    def sua(self):
        print(f"{self.ten}: Gâu gâu!")

class Meo(DongVat):
    def meu(self):
        print(f"{self.ten}: Meo meo!")

# Sử dụng
cho = Cho("Milu")
cho.an()                              # Milu đang ăn... (kế thừa từ DongVat)
cho.sua()                             # Milu: Gâu gâu!   (riêng của Cho)

meo = Meo("Mimi")
meo.an()                              # Mimi đang ăn... (kế thừa)
meo.meu()                             # Mimi: Meo meo!

5. super() — gọi hàm của lớp cha

Khi lớp con có __init__ riêng, ta thường muốn gọi __init__ của lớp cha để không phải viết lại.

class DongVat:
    def __init__(self, ten, tuoi):
        self.ten = ten
        self.tuoi = tuoi

    def thong_tin(self):
        print(f"{self.ten}, {self.tuoi} tuổi")

class Cho(DongVat):
    def __init__(self, ten, tuoi, giong):
        # Gọi __init__ của lớp cha để gán ten + tuoi
        super().__init__(ten, tuoi)

        # Thêm thuộc tính riêng của Cho
        self.giong = giong

    def thong_tin_day_du(self):
        # Gọi thong_tin() của lớp cha
        super().thong_tin()
        print(f"Giống: {self.giong}")

cho = Cho("Milu", 3, "Poodle")
cho.thong_tin_day_du()
# Milu, 3 tuổi
# Giống: Poodle
ℹ️ Tại sao dùng super()?

Không dùng super: phải tự viết self.ten = ten; self.tuoi = tuoi trong Cho.__init__ → code trùng lặp.

Dùng super: gọi super().__init__(ten, tuoi) → code ngắn, nếu lớp cha thay đổi thì lớp con tự cập nhật theo.

6. Ghi đè (override) — đa hình đầu tiên

Ghi đè = lớp con viết lại phương thức của lớp cha (cùng tên, cùng tham số, nhưng thân hàm khác).

class DongVat:
    def __init__(self, ten):
        self.ten = ten

    def keu(self):
        print(f"{self.ten}: ...")     # mặc định không có tiếng

class Cho(DongVat):
    def keu(self):                     # GHI ĐÈ phương thức keu()
        print(f"{self.ten}: Gâu gâu!")

class Meo(DongVat):
    def keu(self):                     # GHI ĐÈ phương thức keu()
        print(f"{self.ten}: Meo meo!")

class Bo(DongVat):
    # Không ghi đè → dùng keu() của DongVat
    pass

# Cùng gọi keu() nhưng kết quả khác nhau → ĐA HÌNH
cho = Cho("Milu")
meo = Meo("Mimi")
bo = Bo("Bò")

cho.keu()    # Milu: Gâu gâu!
meo.keu()    # Mimi: Meo meo!
bo.keu()     # Bò: ...

7. Lớp trừu tượng + abstractmethod

Đôi khi lớp cha muốn ép các lớp con bắt buộc phải ghi đè một phương thức. Dùng ABC + @abstractmethod.

from abc import ABC, abstractmethod

# ABC = Abstract Base Class — lớp trừu tượng, không thể khởi tạo trực tiếp
class HinhHoc(ABC):

    @abstractmethod                  # bộ trang trí ép lớp con phải ghi đè
    def tinh_dien_tich(self):
        pass                          # chỉ khai báo, không có thân

    def gioi_thieu(self):
        print("Tôi là một hình học")

# LỖI — không thể tạo đối tượng từ lớp trừu tượng
# h = HinhHoc()                      # TypeError: Can't instantiate abstract class

# Lớp con BẮT BUỘC phải ghi đè tinh_dien_tich()
class HinhVuong(HinhHoc):
    def __init__(self, canh):
        self.canh = canh

    def tinh_dien_tich(self):         # bắt buộc ghi đè!
        return self.canh * self.canh

class HinhTron(HinhHoc):
    def __init__(self, ban_kinh):
        self.ban_kinh = ban_kinh

    def tinh_dien_tich(self):         # bắt buộc ghi đè!
        return 3.14 * self.ban_kinh ** 2

hv = HinhVuong(5)
print(hv.tinh_dien_tich())            # 25
hv.gioi_thieu()                       # Tôi là một hình học (kế thừa)

ht = HinhTron(3)
print(ht.tinh_dien_tich())            # 28.26
⚠️ Quên ghi đè = lỗi

Nếu lớp con không ghi đè phương thức @abstractmethod, Python sẽ không cho tạo đối tượng:

class HinhTamGiac(HinhHoc):
    def __init__(self, a, b, c):
        self.a, self.b, self.c = a, b, c
    # QUÊN ghi đè tinh_dien_tich()!

htg = HinhTamGiac(3, 4, 5)
# LỖI: Can't instantiate abstract class HinhTamGiac with abstract method tinh_dien_tich

8. Đa hình (polymorphism) — cốt lõi đồ án

Đa hình = "nhiều hình thái". Cùng một lời gọi hàm, nhưng kết quả khác nhau tùy loại đối tượng.

class MatHang:
    def __init__(self, ten, gia):
        self.ten = ten
        self.gia = gia

    def tinh_gia_ban(self):
        return self.gia                # mặc định = giá gốc

class Sach(MatHang):
    def tinh_gia_ban(self):
        return self.gia * 1.10         # Sách: +10% bản quyền

class TapChi(MatHang):
    def tinh_gia_ban(self):
        return self.gia * 1.05         # Tạp chí: +5%

class BaoGiay(MatHang):
    def tinh_gia_ban(self):
        return self.gia                # Báo: không phụ thu

# Tạo list 3 đối tượng khác loại
ds = [
    Sach("Truyện Kiều", 50000),
    TapChi("Tạp chí KH", 30000),
    BaoGiay("Báo Tuổi Trẻ", 5000),
]

# Cùng gọi tinh_gia_ban() — kết quả khác nhau tùy loại!
for sp in ds:
    print(f"{sp.ten}: {sp.tinh_gia_ban():.0f}đ")
# Truyện Kiều: 55000đ
# Tạp chí KH: 31500đ
# Báo Tuổi Trẻ: 5000đ

Đây chính là đa hình: cùng phương thức tinh_gia_ban(), nhưng mỗi lớp con có cách tính riêng.

9. Lớp MatHang trong đồ án

Giờ xem code thực tế trong file loi_thuat_toan/lop_doi_tuong.py:

from abc import ABC, abstractmethod

class MatHang(ABC):
    """
    Lớp CHA trừu tượng đại diện cho một mặt hàng.
    Không thể tạo MatHang() trực tiếp — phải tạo qua lớp con.
    """

    def __init__(self, ma_so, ten_san_pham, gia_co_ban, ton_kho):
        # 4 thuộc tính chung — mọi loại mặt hàng đều có
        self.ma_so = ma_so                  # mã số (PRIMARY KEY)
        self.loai_hang = ''                  # lớp con sẽ ghi đè
        self.ten_san_pham = ten_san_pham    # tên sản phẩm
        # Validate giá không được âm
        if gia_co_ban < 0:
            raise ValueError('Giá cơ bản không được âm.')
        self.gia_co_ban = gia_co_ban
        if ton_kho < 0:
            raise ValueError('Tồn kho không được âm.')
        self.ton_kho = ton_kho

    @abstractmethod                          # ép 5 lớp con phải ghi đè
    def tinh_gia_ban(self):
        """
        Mỗi loại hàng có công thức giá bán riêng:
            Sách      = giá × 1.10  (phụ thu 10% bản quyền)
            Tạp chí   = giá × 1.05  (phụ thu 5%)
            Báo giấy  = giá × 1.00  (không phụ thu)
            Luận văn  = giá × 1.20  (phụ thu 20%)
            Bản thảo  = giá × 1.15  (phụ thu 15%)
        """
        pass

    def chuyen_thanh_dict(self):
        """Chuyển đối tượng thành dict (để dễ gửi ra JSON)."""
        return {
            'ma_so': self.ma_so,
            'loai_hang': self.loai_hang,
            'ten_san_pham': self.ten_san_pham,
            'gia_co_ban': self.gia_co_ban,
            'ton_kho': self.ton_kho,
            'gia_ban': self.tinh_gia_ban()   # ← ĐA HÌNH ở đây!
        }

    def __str__(self):
        return (f'[{self.ma_so}] {self.loai_hang} - {self.ten_san_pham} | '
                f'Giá bán: {self.tinh_gia_ban():.0f}đ | Tồn kho: {self.ton_kho}')
✅ Điểm hay của đa hình

Trong chuyen_thanh_dict(), dòng 'gia_ban': self.tinh_gia_ban() gọi tinh_gia_ban() — nhưng hàm nào được gọi phụ thuộc vào đối tượng là Sach, TapChi, hay lớp con nào.

Cùng 1 dòng code, nhiều kết quả → đó là đa hình!

Lớp Sach (Sách)

class Sach(MatHang):
    """
    Sách — kế thừa MatHang.
    Thuộc tính riêng: tac_gia, nha_xuat_ban.
    Công thức giá bán: gia_co_ban × 1.10 (phụ thu 10% bản quyền).
    """

    def __init__(self, ma_so, ten_san_pham, gia_co_ban, ton_kho, tac_gia, nha_xuat_ban):
        # Gọi __init__ của lớp cha để gán 4 trường chung
        super().__init__(ma_so, ten_san_pham, gia_co_ban, ton_kho)

        # Gán nhãn loại hàng
        self.loai_hang = 'Sách'

        # Gán 2 trường riêng của Sách
        self.tac_gia = tac_gia
        self.nha_xuat_ban = nha_xuat_ban

    def tinh_gia_ban(self):
        """Override — Sách phụ thu 10%."""
        return self.gia_co_ban * 1.1

    def chuyen_thanh_dict(self):
        """Ghi đè để bổ sung 2 trường riêng (tac_gia, nha_xuat_ban)."""
        dict_co_ban = super().chuyen_thanh_dict()   # Lấy dict của cha
        dict_co_ban['tac_gia'] = self.tac_gia        # Thêm trường riêng
        dict_co_ban['nha_xuat_ban'] = self.nha_xuat_ban
        return dict_co_ban

11. Phương thức chuyen_thanh_dict()

Tại sao cần phương thức này? Vì JSON (dùng để gửi dữ liệu qua mạng) chỉ hiểu được dict/list/str/int/float/bool — không hiểu "đối tượng Python". Nên phải chuyển.

# Tạo 3 đối tượng khác loại
sach = Sach("MH001", "Truyện Kiều", 50000, 20, "Nguyễn Du", "NXB Văn học")
tapchi = TapChi("MH002", "Tạp chí KH", 30000, 15, "Số 120")
bao = BaoGiay("MH003", "Báo Tuổi Trẻ", 5000, 50, "2025-06-25")

# Mỗi loại trả dict khác nhau (đa hình)
print(sach.chuyen_thanh_dict())
